    \item Find $<x>, <p>, <x^2>, <p^2>$ and $<T>$, for the nth stationary state of the harmonic oscillator,
    using the method of Example 2.5. Check that the uncertainty principle is satisfied.
    
    \textcolor{hwColor}{
      Starting off with the operators for the harmonic oscillator that are defined as: \\
      \\
      $
        \begin{cases}
          \hat{a}_+=\dfrac{1}{\sqrt{2 m \hbar \omega}}\left(-i \hat{p}+m \omega \hat{x}\right) \Rightarrow \hat{x}=\sqrt{\dfrac{\hbar}{2 m \omega}}(\hat{a}_+ + \hat{a}_-) \\
          \\
          \hat{a}_-=\dfrac{1}{\sqrt{2 m \hbar \omega}}\left(i \hat{p}+m \omega \hat{x}\right) \Rightarrow \hat{x}=i\sqrt{\dfrac{\hbar}{2 m \omega}}(\hat{a}_+ - \hat{a}_-)
        \end{cases} \\
        \\
      $
      We know that the energy of the nth eigenstate is $E_n=\left(n+\dfrac{1}{2}\right) \hbar \omega$ \\
      \\
      $
      \begin{cases}
        \hbar \omega \left(\hat{a}_+\hat{a}_-+\dfrac{1}{2}\right) \psi=E \psi \\
        \\
        \hbar \omega \left(\hat{a}_-\hat{a}_+-\dfrac{1}{2}\right) \psi=E \psi
      \end{cases} \Rightarrow \begin{cases}
        \hbar \omega \left(\hat{a}_+\hat{a}_-+\dfrac{1}{2}\right) \psi_n=\left(n+\dfrac{1}{2}\right) \hbar \omega \psi_n \\
        \\
        \hbar \omega \left(\hat{a}_-\hat{a}_+-\dfrac{1}{2}\right) \psi_n=\left(n+\dfrac{1}{2}\right) \hbar \omega \psi_n 
      \end{cases} \\
      \\
      \\
      \Longrightarrow \begin{cases}
        \hat{a}_+\hat{a}_-\psi_n=n\psi_n \\
        \\
        \hat{a}_-\hat{a}_+\psi_n=\left(n+1\right)\psi_n
      \end{cases} ~~~~ \surd \\ \\
      $
      Now it is time to find $<x>, <p>, <x^2>, <p^2>, <T>$ and $<V>$. \\
      \\
      $
        <x>=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \psi_n(x)\hat{x}dx=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \psi_n(x) \left(\sqrt{\dfrac{\hbar}{2m\omega}}(\hat{a}_+ +\hat{a}_-)\right)dx \\
        \\
        \\
        =\sqrt{\dfrac{\hbar}{2m\omega}} \bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[\hat{a}_+ \psi_n(x)+\hat{a}_- \psi_n(x)\right]dx \\
        \\
        \\
        =\sqrt{\dfrac{\hbar}{2m\omega}} \bigints_{-\infty}^{+\infty} \psi_*(x) \left[A_{n+1} \psi_{n+1}(x)+A_{n-1} \psi_{n-1}(x)\right])dx=0 \\
        \\
        \\
        \therefore ~ <x>=0 ~~~ \surd \\
        \\
        \\
        \\
        \\
        <x^2>=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \psi_n(x)\hat{x}^2 dx=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[\sqrt{\dfrac{\hbar}{2m\omega}}(\hat{a}_+ + \hat{a}_-)\right]^2 \psi_n(x) dx \\
        \\
        \\
        =\dfrac{\hbar}{2m\omega} \bigints_{-\infty}^{+\infty} \psi^*_n(x) \left[\hat{a}^2_+ \psi_n(x)+\hat{a}_+ \hat{a}_- \psi_n(x)+ \hat{a}_- \hat{a}_+ \psi_n(x)+\hat{a}^2_- \psi_n(x)\right]dx \\
        \\
        \\
        \dfrac{\hbar}{2m\omega} \bigints_{-\infty}^{+\infty} \psi^*_n(x) \left[A_{n+2} \psi_{n+2}(x)+n\psi_n(x)+(n+1)\psi_n(x)+A_{n-2}\psi_{n-2}(x)\right]dx \\
        \\
        \\
        \therefore ~ <x^2>=\dfrac{\hbar}{2m\omega} \left[2n+1\right] ~~~ \surd \\
        \\
        \\
        \\
        \\
        <p>=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \psi_n(x)\hat{p} dx=\bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[i\sqrt{\dfrac{\hbar m \omega}{2}}(\hat{a}_+ - \hat{a}_-)\right] \psi_n(x) dx \\
        \\
        \\
        =-i\sqrt{\dfrac{\hbar m \omega}{2}} \bigints_{-\infty}^{+\infty} \psi^{*}_n(x)\left[\hat{a}_+ \psi_n(x)- \hat{a}_- \psi_n(x)\right]dx \\
        \\
        \\
        =-i\sqrt{\dfrac{\hbar m \omega}{2}} \bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[A_{n+1} \psi_{n+1}(x)-A_{n-1} \psi_{n-1}(x)\right] dx \\
        \\
        \\
        =i\sqrt{\dfrac{\hbar m \omega}{2}} \left[A_{n+1} \bigints_{-\infty}^{+\infty} \psi^*_n(x) \psi_{n+1}(x)-A_{n-1} \bigints_{-\infty}^{+\infty} \psi^*_n \psi_{n-1}(x)\right]dx \\
        \\
        \\
        \therefore ~ <p>=0 ~~~ \surd \\
        \\
        \\
        \\
        \\

        \\
        \\
        =-\dfrac{\hbar m \omega}{2} \bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[\hat{a}^2_+ - \hat{a}_+ \hat{a}_- -\hat{a}_- \hat{a}_+ +\hat{a}^2_-\right] \psi_n(x) dx \\
        \\
        \\
        =-\dfrac{\hbar m \omega}{2} \bigints_{-\infty}^{+\infty} \psi^{*}_n(x) \left[A_{n+2} \psi_{n+2}(x)-n \psi_n(x)-(n+1) \psi_n(x)+A_{n-2} \psi_{n-2}(x)\right] dx=-\dfrac{\hbar m \omega}{2} \left[-n-(n+1)\right] \\
        \\
        \\
        \therefore ~ <p^2>=\dfrac{\hbar m \omega}{2}(2n+1) ~~~ \surd \\
        \\
        \\
        \\
        \\
        \begin{cases}
          \sigma_x=\sqrt{<x^2>-<x>^2}=\sqrt{\dfrac{\hbar}{2m \omega}(2n+1)} \\
          \\
          \sigma_p=\sqrt{<p^2>-<p>^2}=\sqrt{\dfrac{\hbar m \omega}{2}(2n+1)}
        \end{cases} \Longrightarrow \sigma_x \sigma_p=\dfrac{\hbar}{2}(2n+1) ~~~ \surd \\ \\
        \\
        \\
        $
        Note that the uncertainty principle is satisfied for the nth eigenstate. ($\sigma_x \sigma_p \geq \dfrac{\hbar}{2}$) \\ \\
        $
        \begin{cases}
          <T>=<\dfrac{p^2}{2m}>=\dfrac{1}{2m}<p^2>=\dfrac{1}{2m}\left(\dfrac{\hbar m \omega}{2}(2n+1)\right)=\dfrac{\hbar \omega}{4}(2n+1) \\
          \\
          <V>=<\dfrac{1}{2}m \omega^2 x^2>=\dfrac{1}{2}m \omega^2<x^2>=\dfrac{1}{2}m \omega^2 \left(\dfrac{\hbar}{2m \omega}(2n+1)\right)=\dfrac{\hbar \omega}{4}(2n+1)
        \end{cases} \\ \\
        \\
        <T>+<V>=E_n
      $
    }

    $$\Psi(x, 0)=A \left[3\psi_0(x)+4\psi_1(x)\right]$$ 
    \begin{enumerate}

      $\psi_1(x)$?

      \item If you measured the energy of this particle, what values might you get, and with what probabilities?

        \textcolor{hwColor}{
          Let's start by normalizing the initial wave function. \\ 
          \\
          $
            1=\bigints_{-\infty}^{+\infty} |\Psi(x,0)|^2 dx=\bigints_{-\infty}^{+\infty} \Psi(x,0) \Psi^*(x,0)dx \\
            \\
            \\
            =\bigints_{-\infty}^{+\infty} A^2\left(3\psi_0(x)+4 \psi_1(x)\right) \left(3 \psi^*_0(x)+4\psi^{*}_1(x)\right)dx=A^2 \bigints_{-\infty}^{+\infty} \left(3\psi_0(x)+4 \psi_1(x)\right) \left(3 \psi^*_0(x)+4\psi^{*}_1(x)\right)dx \\
            \\
            \therefore ~ A=\dfrac{1}{5} \surd \\ \\
          $
          so now we can have the wave function as: \\
          \\
          $
            \Psi(x,0)=\dfrac{3}{5} \psi_0(x)+\dfrac{4}{5} \psi_1(x) \\
            \\
            \\
            \therefore ~ \Psi(x,0)=\dfrac{3}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right)+\dfrac{4}{5}x \left(\dfrac{4m^3 \omega^3}{\pi \hbar}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar} x^2\right) \\ 
            \\
          $
          The general solution to the Schr$\ddot{o}$dinger equation with a harmonic oscillator potential: \\
          \\
          $
            \Psi(x,t)=B_0 \psi_0(x) \phi_0(t)+B_1 \psi_1(x) \phi_1(t)+B_2 \psi_2(x) \phi_2(t)+... \\
            \\
            =B_0 \psi_0(x) exp\left(-\dfrac{i\omega t}{2}\right)+B_1 \psi_1(x) exp\left(-\dfrac{i\omega t}{2}\right)+B_2 \psi_2(x) exp\left(-\dfrac{i\omega t}{2}\right)+... \\
            \\
            =B_0 \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} ~ exp \left(-\dfrac{m \omega}{2\hbar}x^2\right)+B_1 x \left(\dfrac{4m^3 \omega^3}{\pi \hbar^3}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right) ~ \exp\left(-3i\omega t/2\right)+B_2 \left(\dfrac{m \omega}{4 \pi \hbar}\right)^{1/4} \left(\dfrac{2 m \omega}{\hbar}x^2-1\right) ~ exp\left(-\dfrac{m\omega}{2\hbar}x^2\right)e^{-5i \omega t/2}+... \\
            \\
            \\
          $
          By setting $t=0$, we have: \\
          \\
          $
            \Psi(x,t)=B_0 \psi_0(x)+B_1 \psi_1(x)+B_2 \psi_2(x)+... \\ \\
            =B_0 \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right)+B_1 x \left(\dfrac{4 m^3 \omega^3}{\pi \hbar^3}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right)+B_2 \dfrac{m \omega}{4 \pi \hbar}^{1/4} \left(\dfrac{2 m \omega}{\hbar}x^2-1\right) ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right)+... \\
            \\
            \\
            \Longrightarrow \begin{cases}
              B_0=\dfrac{3}{5} \\
              \\
              B_1=\dfrac{4}{5} \\
              \\
              B_n=0, ~~~ n\geq 2
            \end{cases} \\ \\ 
          $
          The general solution. \\
          \\
          $
            \Psi(x,t)=\dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left(3 ~ exp(-i \omega t/2)+4\sqrt{\dfrac{2 m \omega}{\hbar}}x ~ exp(-3 i\omega t/2)\right) ~ exp \left(-\dfrac{m \omega}{2\hbar}x^2\right) \\
            \\
            \\
          $
          Now for the probabilities of measuring, we have $E_0=\dfrac{\hbar \omega}{2}$ and $E_1=\dfrac{3 \hbar \omega}{2}$. \\
          \\
          \\
          $
            \Psi(x,t)=\dfrac{3}{5} \psi_0(x) ~ exp\left(-i E_0 t/\hbar\right)+\dfrac{4}{5} \psi_1(x) ~ exp\left(-i E_1 t/ \hbar\right) \Rightarrow \begin{cases}
              P(E_0)=|\dfrac{3}{5}|^2 \\
              \\
              P(E_1)=|\dfrac{4}{5}|^2
            \end{cases}
          $ \\ \\ \\
          Now that the wave function is known, the probability distribution for the particle’s position at
          time t can be found: \\
          \\
          $
            |\Psi(x,t)|^2=\Psi(x,t) \Psi^*(x,t)= \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left(3 ~ exp(-i \omega t/2)+4\sqrt{\dfrac{2 m \omega}{\hbar}}x ~ exp(-3 i\omega t/2)\right) ~ exp \left(-\dfrac{m \omega}{2\hbar}x^2\right) 
            \times \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left(3 ~ exp(i \omega t/2)+4\sqrt{\dfrac{2 m \omega}{\hbar}}x ~ exp(3 i\omega t/2)\right) ~ exp \left(-\dfrac{m \omega}{2\hbar}x^2\right) \\
            \\
            \\
            \therefore ~ |\Psi(x,t)|^2=\dfrac{1}{25}\sqrt{\dfrac{m \omega}{\pi \hbar}}\left(\dfrac{32 m \omega}{\hbar}+24\sqrt{\dfrac{2m \omega}{\hbar}}x cos(\omega t)+9\right) ~ exp\left(-\dfrac{m \omega}{\hbar}x^2\right) ~~~ \surd \\ \\
          $
          If the provided initial condition were instead: \\
          \\
          $
            \Psi(x,0)=A\left(3 \psi_0(x)+4\psi_2(x)\right) \Longrightarrow A=\dfrac{1}{5} \Longrightarrow \Psi(x,0)=\dfrac{3}{5} \psi_0(x)+\dfrac{4}{5} \psi_2(x) \\ \\
          $
          Then the general solution: \\
          \\
          \\
          $
            \Psi(x,0)=\dfrac{3}{5} \psi_0(x) ~ exp\left(-iE_0 t/\hbar\right)+\dfrac{4}{5} \psi_2(x) ~ exp\left(-iE_2 t/\hbar\right) \\ 
            \\
            \\
            \therefore ~ \Psi(x,t)=\dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left[3 e^{-i \omega t/2}+\dfrac{4}{\sqrt{2}}\left(\dfrac{2m\omega}{\hbar}x^2-1\right)e^{-5i \omega t/2}\right]~ exp\left(-\dfrac{m \omega }{2\hbar}x^2\right) \surd \\
            \\
            \\
            \\
            |\Psi(x,t)|^2=\Psi(x,t) \Psi^*(x,t)= \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left[3 e^{-i \omega t/2}+\dfrac{4}{\sqrt{2}}\left(\dfrac{2m\omega}{\hbar}x^2-1\right)e^{-5i \omega t/2}\right]~ exp\left(-\dfrac{m \omega }{2\hbar}x^2\right) 
            \times \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left[3 e^{i \omega t/2}+\dfrac{4}{\sqrt{2}}\left(\dfrac{2m\omega}{\hbar}x^2-1\right)e^{5i \omega t/2}\right]~ exp\left(-\dfrac{m \omega }{2\hbar}x^2\right) \\
            \\
            \\
            \therefore ~ |\Psi(x,t)|^2=\dfrac{1}{25} \sqrt{\dfrac{m \omega}{\pi \hbar}} \left[9+8\left(\dfrac{2m\omega}{\hbar}x^2-1\right)^2+12\sqrt{2} \left(\dfrac{2m\omega}{\hbar}x^2-1\right) cos(2\omega t)\right] ~ exp\left(-\dfrac{m \omega}{\hbar}x^2\right) \surd \\
            \\
            \\
            \\
          $
          Now we can find $<x>, <p>$: \\
          \\
          \\
          $
            <x>=\bigints_{-\infty}^{+\infty} \Psi^*(x,t) \Psi(x,t) x dx=\bigints_{-\infty}^{+\infty} |\Psi(x,t)|^2 dx \\
            \\
            =\bigints_{-\infty}^{+\infty} \dfrac{1}{25} \sqrt{\dfrac{m \omega}{\pi \hbar}}x \left(9+\dfrac{32 m\omega}{\hbar}x^2+24\sqrt{\dfrac{2m\omega}{\hbar}}x cos(\omega t)\right) ~ exp\left(-\dfrac{m\omega}{\hbar}x^2\right) dx \\
            \\
            =\dfrac{24}{25} \sqrt{\dfrac{2}{\pi}} cos(\omega t) \bigints_{-\infty}^{+\infty}\dfrac{m \omega}{\hbar} x^2 ~ exp\left(-\dfrac{m \omega}{\hbar}x^2\right)dx \\
            \\
            \\
            =\dfrac{24}{25} \sqrt{\dfrac{2}{\pi}} cos(\omega t) \bigints_{-\infty}^{+\infty} \xi^2 ~ exp\left(-\xi^2\right) \sqrt{\dfrac{\hbar}{m \omega}} d\xi \\ 
            \\
            \\
            \therefore ~ <x>=\dfrac{12}{25} \sqrt{\dfrac{2\hbar}{m\omega}} cos(\omega t) ~~~ \surd \\
            \\
          $
          According to Ehrenfest’s theorem: \\
          \\
          $
            <p>=m<v>=m\dfrac{d<x>}{dt}=m \dfrac{d}{dt} \left(\dfrac{12}{25} \sqrt{\dfrac{2\hbar}{m\omega}} cos(\omega t)\right)=-\dfrac{12}{25} \sqrt{2 \hbar m \omega} sin(\omega t) \\
            \\
            \\
            \\
          $
          Check this result by calculating the expectation value of $p$ at time $t$ manually. \\
          \\
          $
            <p>=\bigints_{-\infty}^{+\infty}  \Psi^*(x,t) \left(-i \hbar \dfrac{\partial}{\partial x}\right) \Psi(x,t) dx= \\
            \\
            =-i \hbar \bigints_{-\infty}^{+\infty} \Psi^*(x,t) \dfrac{\partial \Psi}{\partial x} dx \\
            \\
            \\
            =-i \hbar \bigints_{-\infty}^{+\infty} \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left(3 e^{i \omega t/2}+4\sqrt{\dfrac{2m \omega}{\hbar}}x e^{3 i \omega t/2}\right) ~ exp\left(-\dfrac{m \omega}{2\hbar} x^2\right) \\ \\
            \times \dfrac{\partial }{\partial x} \dfrac{1}{5} \left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} \left(3 e^{-i \omega t/2}+4\sqrt{\dfrac{2m \omega}{\hbar}}x e^{3 -i \omega t/2}\right) ~ exp\left(-\dfrac{m \omega}{2\hbar} x^2\right) \\
            \\
            \\
            \\
            \therefore ~~ <p>=-\dfrac{12}{25} \sqrt{2 \hbar m \omega} \omega cos(\omega t) ~~~ \surd \\
          $ 
            \\
            \\
            Now let's see if $\dfrac{d<p>}{dt}=<-\dfrac{\partial V}{\partial x}>$ is true or not. \\
            \\
            \\
          $
            \dfrac{d}{dt} \left(-\dfrac{12}{25} \sqrt{2 \hbar m \omega} sin(\omega t)\right)=<-\dfrac{\partial}{\partial}\left(\dfrac{1}{2} m \omega^2 x^2\right)> \\
            \\
            \therefore ~~ -\dfrac{12}{25} \sqrt{2 \hbar m \omega} \omega cos(\omega t)=-\dfrac{12}{25} \sqrt{2 \hbar m \omega} \omega cos(\omega t) ~~~ \surd
          $
        }

    \end{enumerate}

  \textbf{2.14}
  \begin{itemize}
    \item In the ground state of the harmonic oscillator, what is the probability (correct to three
    significant digits) of finding the particle outside the classically allowed region?
    \emph{Hint:} Classically, the energy of an oscillator is $E=\dfrac{1}{2}ka^2=\dfrac{1}{2}m \omega a^2$,
    where $a$ is the amplitude. So the "classically allowed region" for an oscillator of energy $E$
    extends from $-\sqrt{\dfrac{2E}{m \omega^2}}$ to $+\sqrt{\dfrac{2E}{m \omega^2}}$. Look in math table
    under "Normal Distribution" or "Error Function" for the numerical value of the integral, or evaluate
    it by computer.

      \textcolor{hwColor}{
        Born's interpretation, says that the probability distribution for the particle’s position in the
        ground state at time t is given by $|\Psi_0(x,t)|^2$. Classically, the probability that the particle 
        lies in the allowed region is \\ \\ 
        $$\bigints_{-a}^{+a} |\Psi_0(x,t)|^2 dx$$ \\
        \\
        The probability that it’s not in this region is: \\
        \\
        $
          1-\bigints_{-a}^{+a} |\Psi_0(x,t)|^2 dx=1-\bigints_{-a}^{+a} \Psi_0(x,t) \Psi^*_0(x,t) dx=1-\bigints_{-a}^{+a} \left[\Psi_0(x,t) e^{-i E_0 t/\hbar}\right] \left[\Psi^*_0(x,t) e^{i E_0 t/\hbar}\right] dx \\
          \\
          \\
          =1-\bigints_{-a}^{+a} |\Psi_0(x,t)|^2 dx=1-\bigints_{-a}^{+a} \left[\left(\dfrac{m \omega}{\pi \hbar}\right)^{1/4} ~ exp\left(-\dfrac{m \omega}{2\hbar}x^2\right)\right]^2 dx \\
          \\
          \\
          =1-\sqrt{\dfrac{m \omega}{\pi \hbar}} \bigints_{-a}^{+a} ~ exp\left(-\dfrac{m \omega}{\hbar} x^2\right)dx  \\ \\ 
        $
        Like the previous problems using $\xi \equiv \sqrt{\dfrac{m \omega}{\hbar}x}$ defined variable helps us to solve this problem faster. \\
        \\
        \\
        $
          1-\bigints_{-a}^{+a} |\Psi_0(x,t)|^2 dx=1-\sqrt{\dfrac{m \omega}{\pi \hbar}} \bigint_{-\sqrt{\dfrac{m \omega}{\hbar}a}}^{+\sqrt{\dfrac{m \omega}{\hbar}a}} ~ \sqrt{\dfrac{\hbar}{m \omega}} ~ exp(-\xi^2) d\xi \\
          \\
          \\
          =1-\dfrac{2}{\sqrt{\pi}} \bigint_{0}^{+\sqrt{\dfrac{m \omega}{\hbar}a}} exp(-\xi^2) d\xi \\
          \\
        $
        The error function is $erf ~ z=\dfrac{2}{\sqrt{\pi}} \bigint_{0}^{z} exp(-x^2) dx$. So now we can rewrite 
        the probability as the following: \\
        \\
        $
          1-\bigints_{-a}^{+a} |\Psi_0(x,t)|^2 dx=1-erf(\sqrt{\dfrac{m \omega}{\hbar}a}) \\ \\
        $
        Now we cand find $a$ easily: $\dfrac{\hbar \omega}{2}=\dfrac{1}{2} m \omega^2 a^2 \Longrightarrow a=\sqrt{\dfrac{\hbar}{m \omega}}$.
        Therefore, the probability that the particle lies outside the allowed region in the ground state is: \\
        \\
        $
          1-erf(1) \approx 0.16
        $
      }

  \end{itemize}

  \textbf{2.44}
  \begin{itemize}
    \item If two (or more) distinct solutions to the (time-independent) Schr$\ddot{o}$dinger equation have the
    same energy $E$, these states are said to be \textbf{degenerate}. For example the free particle states are 
    doubly degenerate -- one solution representing motion to the right, and the other motion to the left. But
    we have never encountered \emph{normalizable} degenerate solutions, and this is no accident. Prove the following,
    theorem: \emph{In one dimension ($-\infty < x < +\infty$) there are no degenerate bound states. [Hint:} suppose
    there are \emph{two} solutions, $\psi_1$ and $\psi_2$, with the same energy $E$. Multiply the Schr$\ddot{o}$dinger equation
    for $\psi_1$ by $\psi_2$, and the Schr$\ddot{o}$dinger equation for $\psi_2$ by $\psi_1$, and subtract,
    to show that $\left(\psi_2 \dfrac{d\psi_1}{dx}-\psi_1 \dfrac{d\psi_2}{dx}\right)$ is a constant. Use the 
    fact that for normalizable solutions $\psi \longrightarrow 0$ at $\pm \infty$ to demonstrate that this constant
    is in fact zero. Conclude that $\psi_2$ is a multiple of $\psi_1$, and hence that the two solutions
    are not distinct.]

      \textcolor{hwColor}{
        Let's start with the one-dimensional Schr$\ddot{o}$dinger equation: \\
        \\
        $
          i\hbar \dfrac{\partial \Psi}{\partial t}=-\dfrac{\hbar^2}{2m}\dfrac{\partial^2 \Psi}{\partial x^2}+V(x,t)\Psi(x,t) \\
          \\
          \\
          \begin{cases}
            i\hbar \dfrac{\phi^'(t)}{\phi(t)}=E \\
            \\
            -\dfrac{\hbar^2}{2m}\dfrac{\psi^{''}(x)}{\psi(x)}+V(x)=E
          \end{cases} \\
          \\ \\
        $
        This ODE in x is the time-independent Schr$\ddot{o}$dinger equation and can be written as: \\
        \\
        $
          \dfrac{\partial^2 \psi}{\partial x^2}=\dfrac{2m}{\hbar} \left[V(x)-E\right]\psi \\
          \\
          \\
          \begin{cases}
            \dfrac{d^2 \psi_1}{dx^2}=\dfrac{2m}{\hbar^2}\left[V(x)-E\right]\psi_1 \\
            \\
            \dfrac{d^2 \psi_2}{dx^2}=\dfrac{2m}{\hbar^2}\left[V(x)-E\right]\psi_2
          \end{cases} \Rightarrow \begin{cases}
            \psi_2 \dfrac{d^2 \psi_1}{dx^2}=\dfrac{2m}{\hbar^2}\left[V(x)-E\right]\psi_1 \psi_2 \\
            \\
            \psi_1 \dfrac{d^2 \psi_2}{dx^2}=\dfrac{2m}{\hbar^2}\left[V(x)-E\right]\psi_2 \psi_2
          \end{cases} \\
          \\
          \\
          \psi_2 \dfrac{d^2 \psi_1}{dx^2}-\psi_1 \dfrac{d^2 \psi_2}{dx^2}=0 \\
          \\
          \\
          \psi_2 \dfrac{d^2 \psi_1}{dx^2}+\dfrac{d \psi_2}{dx} \dfrac{d \psi_1}{dx}-\psi_1 \dfrac{d^2 \psi_2}{dx^2}-\dfrac{d \psi_1}{dx} \dfrac{d \psi_2}{dx}=0 \\
          \\
          \\
          \dfrac{d}{dx}\left(\psi_2 \dfrac{\psi_1}{dx}\right)-\dfrac{d}{dx}\left(\psi_1 \dfrac{d\psi_2}{dx}\right)=0 \\ \\
        $
        Let's take integral of both sides now: \\
        \\
        $
          \psi_2 \dfrac{d\psi_1}{dx}-\psi_1 \dfrac{d \psi_2}{dx}=C \\ \\
        $
        What we found now must hold for any value of $x$, so set $x=\infty$ to find $C$. \\
        \\
        $
          \psi_2 \dfrac{d\psi_1}{dx}\Big|_{x=\infty}-\psi_1 \dfrac{d \psi_2}{dx}\Big|_{x=\infty}=C \\ \\
        $
        The boundary conditions say $\psi_1(x)=0$ and $\psi_2(x)=0$ when $x\longrightarrow \pm \infty$, therefore $C=0$. \\ \\  
        $
          \Longrightarrow \psi_2 \dfrac{d\psi_1}{dx}-\psi_1 \dfrac{d \psi_2}{dx}=0 \rightarrow \dfrac{d}{dx}ln \psi_1(x)=\dfrac{d}{dx}ln \psi_2(x) \\
          \\
          \\
          \therefore ~~ ln \psi_1(x)=ln \psi_2(x)+W \\
          \\
          \\
          \psi_1(x)=e^{ln \psi_2(x)+W} \\
          \\
          \psi_1(x)=\psi_2(x) e^W \\
          \\
          \psi_1(x)=A\psi_2(x) \\ \\
        $
        Therefore, there are no degenerate bound states in one dimension. To elaborate on this, $\psi_1$ is a constant multiple 
        of $\psi_2(x)$. This is denying the assumption that $\psi_1(x)$ and $\psi_2(x)$ are linearly independent.  
      }

  \end{itemize}
